## Install

```bash
npm install --save dynamo
```

## Usage

```jsx
import React, { Component } from 'react'

import DynoBuilder from 'dynamo'
import 'dynamo/dist/index.css'

class Example extends Component {
  render() {
    return <DynoBuilder />
  }
}
```
